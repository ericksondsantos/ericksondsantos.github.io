var heading = document.firstElementChild.lastElementChild.firstElementChild
    <h1>Hello</h1>
heading.innerHTML = "Good Bye";
    Changes the h1 text to Good Bye
heading.style.color = "Red";
    Changes the color of the h1 text to Red

document.querySelector("input").click();
    Simulates a mouse click

document.firstElementChild.lastElementChild.lastElementChild.lastElementChild.innerHTML = "Eric";
    Changes the li Third to Eric

document.getElementsByTagName("li");
    [li.item, li.item, li.item]
document.getElementsByTagName("li")[2].style.color = "Purple";
    Changes the color of the 3rd li item (index 2) to Purple

document.getElementsByClassName("btn");
    [button.btn]

document.getElementById("heading");
    <h1 id="heading">Hello</h1>
document.getElementById("heading").innerHTML = "Good Bye"
    Changes the h1 text to Good Bye

document.querySelector("<Can be an html tag, #id, or .class>");
document.querySelector("h1");
    <h1 id="heading">Good Bye</h1>
document.querySelector("#heading");
    <h1 id="heading">Good Bye</h1>

document.querySelectorAll("#list .item");
    [li.item, li.item, li.item]

document.querySelectorAll("#list .item")[1].style.color = "Blue";
    Changes the color of the 2nd li item (index 1) to Blue

document.querySelector(".btn").style.backgroundColor = "Yellow";
    Changes the color of the btn class to Yellow

document.querySelector("button").classList.add("invisible");
    Adds the class invisible to <button>
    <button class="btn invisible">Click Me</button>

document.querySelector("button").classList.remove("invisible");
    Removes the class invisible to <button>
    <button class="btn">Click Me</button>

document.querySelector("button").classList.toggle("invisible");
    Adds or Removes the class invisible to <button>

document.querySelector("h1").innerHTML = "<em>Good Bye</em>";
    Use HTML tags inside .innerHTML

document.querySelector("a")
    <a href="https://wwww.bing.com">Google</a>
document.querySelector("a").attributes;
    {0: href, href: href, length: 1}
document.querySelector("a").getAttribute("href");
    https://www.google.com
document.querySelector("a").setAttribute("href", "https://www.bing.com);
    Changes the href from https://www.google.com to https://www.google.com

document.querySelector(".w").classList;
    ['w', 'drum', value: 'w drum']
document.querySelector(".w").classList.add("pressed");
    Adds 'pressed' to the class
document.querySelector(".w").classList;
    ['w', 'drum', 'pressed', value: 'w drum pressed']
document.querySelector(".w").classList.remove("pressed");
    Removes 'pressed' from the class
